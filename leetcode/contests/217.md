class Solution {
public:
    vector<int> ans;
    void solve(vector<int>&nums, int start, int end, int k){
        if (k == 0){
            return ;
        }
//         find min in [start, end-(k-1)]
        int this_min_index = start;
        
        for (int i = start; i < end-(k-1); i++){
            if (nums[i] < nums[this_min_index]){
                this_min_index = i;
            }
        }
        ans.push_back(nums[this_min_index]);
        solve(nums, this_min_index+1, end, k-1);
    }
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        solve(nums, 0, nums.size(), k);

};

class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {
        int start = 0;
        int end = nums.size();
        if(k == end){
            return nums;
        }
        bool isSorted = true;
        for (int i = 1; i < end; i++){
            if (nums[i-1] > nums[i]){
                isSorted = false;
                break;
            }
        }


        vector<int> ans;

        if (isSorted){
            for (int i = 0; i < k; i++){
                ans.push_back(nums[i]);
            }
            return ans;
        }

        for (int i = k-1; i >= 0; i--){
            int min_index = start;
//             find min in [start, end-i]
            for (int j = start; j < end-i; j++){
                if (nums[j] < nums[min_index]){
                    min_index = j;
                }
            }
            ans.push_back(nums[min_index]);
            start = min_index+1;
        }
        return ans;
    }
};

class Solution {
public:
    int minMoves(vector<int>& nums, int limit) {
        int len = nums.size();
        int ans = nums.size();
        for (int i = limit*2; i >= 2; i--){
//             sum is i
            int sum = i;
            int cur_value = 0;
            for (int j = 0; j < len/2; j++){
//                 nums[j]  nums[len-1-j]
                if (nums[j]+nums[len-1-j] != sum){
                    if (max(nums[j], nums[len-1-j])+limit < sum){
                        cur_value += 2;
                    } else if (min(nums[j], nums[len-1-j])+1 > sum) {
                        cur_value += 2;
                    } else {
                        cur_value += 1;
                    }
                }
            }
            
            if (cur_value < ans){
                ans = cur_value;
            }
        }
        return ans;
    }
};

class Solution {

    int minMoves(vector<int>& nums, int limit) {
        int len = nums.size();

        vector<int> sums(len/2);
        vector<int> maxs(len/2);
        vector<int> mins(len/2);
        for (int i = 0; i < len/2; i++){
            sums[i] = nums[i]+nums[len-1-i];
            maxs[i] = max(nums[i], nums[len-1-i]);
            mins[i] = min(nums[i], nums[len-1-i]);

//             sum is i

//                 nums[j]  nums[len-1-j]

                        cur_value += 2;
                    } else if (mins[j]+1 > sum) {
                        cur_value += 2;
                    } else {
                        cur_value += 1;
                    }
                }
            }
