```cpp
#include <vector>
using namespace std;
// @lc code=start
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int len = nums.size();
        if (len == 0){
            return vector<int>{-1,-1};
        }
        if (len == 1){
            if (nums[0] == target){
                return vector<int>{0,0};
            }
            return vector<int>{-1,-1};
        }
        int ans_mid = -1;
        // make sure that len >= 2
        int l = 0;
        int r = len;
        int mid;
        // l < target; r >= target
        while(l < r){
            mid = (l+r)/2;
            if (nums[mid] > target){
                r = mid;
            } else if (nums[mid] < target){
                l = mid+1;
            } else if (nums[mid] == target){
                ans_mid = mid;
                break;
            }
        }
        if (ans_mid == -1){
            return vector<int>{-1,-1};
        }
        // find the target
        int ans_first = ans_mid;
        int ans_last = ans_mid;
        while ( ans_first-1 >= 0 && nums[ans_first-1] == target){
            ans_first--;
        }
        while (ans_last+1 <= len-1 && nums[ans_last+1] == target){
            ans_last++;
        }
        return vector<int>{ans_first, ans_last};
    }
};
```

```cpp
            if (nums[mid] == target){
                r = mid;
            } else if (nums[mid] < target){
                l = mid+1;
            } else if (nums[mid] > target){
                r = mid;
            }
        }
        if (l >= len || nums[l]!= target){
            return vector<int>{-1,-1};
        }
        // find the target
        int ans_first = l;
        int ans_last;
        // right bound
        l = 0;r = len;
        while (l < r){
            mid = (l+r)/2;
            if (nums[mid] == target){
                l = mid + 1;
            } else if (nums[mid] < target){
                l = mid + 1;
            } else if (nums[mid] > target){
                r = mid;
            }
        }
        ans_last = l-1;
        return vector<int>{ans_first, ans_last};
    }
};
```

```cpp
#include <vector>
using namespace std;
// @lc code=start
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        int len = nums.size();
        int ans_first = left_bound(nums, target);
        int ans_last = right_bound(nums, target);
        return vector<int>{ans_first, ans_last};
    }
    int left_bound(vector<int> & nums, int target) {
        int left = 0, right = nums.size();;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] < target) {
                left = mid + 1;
            } else if (nums[mid] > target) {
                right = mid;
            } else if (nums[mid] == target) {
                // 别返回，锁定左侧边界
                right = mid;
            }
        }
        // 最后要检查 left 越界的情况
        if (left >= nums.size() || nums[left] != target)
            return -1;
        return left;
    }

    int right_bound(vector<int> & nums, int target) {
        int left = 0, right = nums.size();
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (nums[mid] < target) {
                left = mid + 1;
            } else if (nums[mid] > target) {
                right = mid;
            } else if (nums[mid] == target) {
                // 别返回，锁定右侧边界
                left = mid + 1;
            }
        }
        // 最后要检查 right 越界的情况，另外注意right需要减1
        if (right-1 < 0 || nums[right-1] != target)
            return -1;
        return right-1;
    }
};****
```
